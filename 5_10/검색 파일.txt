	<sql id="searchCondition">
		<choose>
			<when test='option=="T"'>
				AND title LIKE concat('%', #{keyword}, '%')
			</when>
			<when test='option=="W"'>
				AND writer LIKE concat('%', #{keyword}, '%')
			</when>
			<otherwise>
				AND (title LIKE concat('%', #{keyword}, '%') OR content LIKE concat('%', #{keyword}, '%'))
			</otherwise>
		</choose>
	</sql>
